DEFAULT_CONFIG_HELP = 'Set the key for the default configuration' module-attribute

Help text for the default configuration key option.

DEFAULT_CONFIG_KEY = 'default' module-attribute

Default configuration key for Databricks jobs.

DEFAULT_CONF_FOLDER = 'conf' module-attribute

Default folder for Kedro configurations.

DEFAULT_PROVIDER = 'azure' module-attribute

Default cloud provider for Databricks Asset Bundle.

DEFAULT_TARGET = 'dev' module-attribute

Default target environment for Databricks configurations.

GITIGNORE = f'# Kedro Databricks.databricksconf/{DEFAULT_TARGET}/**!conf/{DEFAULT_TARGET}/.gitkeep'.strip() module-attribute

Content to be added to .gitignore for Kedro Databricks configurations.

IGNORED_OVERRIDE_KEYS = ['init_scripts', 'access_control_list'] module-attribute

Keys that should be ignored when overriding the Databricks configuration.

INVALID_CONFIG_MSG = '\nNo `databricks.yml` file found. Maybe you forgot to initialize the Databricks bundle?\n\nYou can initialize the Databricks bundle by running:\n\n```\nkedro databricks init\n```\n' module-attribute

Message displayed when no databricks.yml file is found in the project.

KEDRO_VERSION = [int(x) for x in kedro_version.split('.')] module-attribute

Kedro version used to build this plugin.

MINIMUM_DATABRICKS_VERSION = [0, 205, 0] module-attribute

Minimum Databricks version required for this plugin.

NODE_TYPE_MAP = {'aws': 'm5.xlarge', 'azure': 'Standard_DS3_v2', 'gcp': 'n1-standard-4'} module-attribute

Map of node types for different cloud providers.

OVERRIDE_KEY_MAP = {'job_clusters': 'job_cluster_key', 'tasks': 'task_key', 'depends_on': 'task_key', 'environments': 'environment_key', 'on_start': 'id', 'on_success': 'id', 'on_failure': 'id', 'on_duration_warning_threshold_exceeded': 'id', 'on_streaming_backlog_exceeded': 'id', 'rules': 'metric', 'parameters': 'name', 'libraries': 'whl', 'subscriptions': 'user_name'} module-attribute

Map of keys that should be overridden in the Databricks configuration.

PROVIDER_PROMPT = '\nPlease select your cloud provider:\n- azure\n- aws\n- gcp\n' module-attribute

Prompt for selecting the cloud provider for Databricks Asset Bundle.

TASK_KEY_ORDER = ['task_key', 'job_cluster_key', 'new_cluster', 'depends_on', 'spark_python_task', 'python_wheel_task'] module-attribute

Order of keys in the task configuration for Databricks jobs.

WORKFLOW_KEY_ORDER = ['name', 'tags', 'access_control_list', 'email_notifications', 'schedule', 'max_concurrent_runs', 'job_clusters', 'tasks'] module-attribute

Order of keys in the workflow configuration for Databricks jobs.
